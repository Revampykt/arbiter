import tracemalloc
import re
import sys
import time

if __name__ == '__main__':
    tracemalloc.start()

    startTime = time.time()

    before = tracemalloc.take_snapshot()

    performSolution(sys.argv)

    after = tracemalloc.take_snapshot()

    tracemalloc.stop()

    stopTime = time.time()

    elapsedTime = stopTime - startTime

    stats = after.compare_to(before, 'lineno')

    usedMemory = 0

    for stat in stats:
        strstat = str(stat)
        size = re.findall('size=([0-9]*)', strstat)[0]
        usedMemory += int(size)

    print('UsedMemory =', usedMemory, ' ElapsedTime =', elapsedTime)
